# View

```tsx
import { View } from '@brain-game/bgui';
import { Text } from '@brain-game/bgui';

export default function ViewExample() {
  return (
    <View style={{ padding: 16 }}>
      <Text>Content</Text>
    </View>
  );
}
```

## Simple Usage

```tsx
<View style={{ padding: 16 }}>
  <Text>Content</Text>
</View>
```
